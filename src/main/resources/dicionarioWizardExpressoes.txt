  ARITH_OP      =: + | - | * | /
  BOOL_OP       =: AND | OR
  ARITH_COMP_OP =: BASIC_COMP_OP | GREATER THAN OR EQUAL TO | LESS THAN OR EQUAL TO | GREATER THAN | LESS THAN
  BASIC_COMP_OP =: EQUAL TO | DIFFERENT THAN
  
  VALUE =: ARITHMETIC | STRING | BOOLEAN | DATE | TIMESPAN | TRANSITION | ENUMERATION
  
  ARITHMETIC =: ( ARITHMETIC ARITH_OP ARITHMETIC ) | [NUMBER_VAR] | NUMBER_CONSTANT | BOOLEAN ? ARITHMETIC : ARITHMETIC
  
  STRING =: [STRING_VAR] | "STRING_CONSTANT" | VALUE + STRING | STRING + VALUE | BOOLEAN ? STRING : STRING
  
  BOOLEAN =: true | false | [BOOLEAN_VAR] | ( ARITHMETIC ARITH_COMP_OP ARITHMETIC ) | ( ARITHMETIC INARITH_COMP_OP ARITHMETIC ) |
              | ( STRING BASIC_COMP_OP STRING ) | (DATE ARITH_COMP_OP DATE) | ( BOOLEAN BOOL_OP BOOLEAN ) | NOT BOOLEAN
  
  DATE =: DATE_CONSTANT | [DATE_VAR] | DATE + TIMESPAN | DATE - TIMESPAN | BOOLEAN ? DATE : DATE
  
  TIMESPAN =: ARITHMETIC TIME_MEASURE
  
  TIME_MEASURE =: DAY | MONTH | YEAR

  TRANSITION =: TRANSITION_VALUE | BOOLEAN ? TRANSITION : TRANSITION

  ENUMERATION =: ENUMERATION_VALUE | BOOLEAN ? ENUMERATION : ENUMERATION

  FORM =: STRING

  PAGE =: STRING

  FRAME =: STRING
  
  TASK_PAGE =: STRING
